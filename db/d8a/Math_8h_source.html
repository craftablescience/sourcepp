334  // Convert from 16-bit (or 15-bit) integers to floating point values in the range [-1, 1]
335  const float fx = (static_cast<float>(this->x) / 32767.5f) - 1.f; // x / ((2^16 - 1) / 2) - 1
336  const float fy = (static_cast<float>(this->y) / 32767.5f) - 1.f; // y / ((2^16 - 1) / 2) - 1

338

344  fw = -fw;
