35bool isAppUsingSource2EnginePredicate(std::string_view installDir) {

66template<bool(*P)(std::string_view)>
67bool isAppUsingEngine(const Steam* steam, AppID appID) {
68 static std::unordered_set<AppID> knownIs = ::getAppsKnownToUseEngine(P);
69 if (knownIs.contains(appID)) {
70 return true;
71 }
72
73 static std::unordered_set<AppID> knownIsNot;
74 if (knownIsNot.contains(appID)) {
75 return false;
76 }
77
78 if (!steam->isAppInstalled(appID)) {
79 return false;
80 }
81
82 const auto installDir = steam->getAppInstallDir(appID);
83 if (std::error_code ec; !std::filesystem::exists(installDir, ec)) [[unlikely]] {
84 return false;
85 }
86
87 if (P(installDir)) {
88 knownIs.emplace(appID);
89 return true;
90 }
91 knownIsNot.emplace(appID);
92 return false;
93}

97Steam::Steam() {
98 std::filesystem::path steamLocation;
99 std::error_code ec;
100
101#ifdef _WIN32
102 {
103 // 16383 being the maximum length of a path
104 static constexpr DWORD STEAM_LOCATION_MAX_SIZE = 16383;
105 std::unique_ptr<char[]> steamLocationData{new char[STEAM_LOCATION_MAX_SIZE]};
106
107 HKEY steam;
108 if (RegOpenKeyExA(HKEY_LOCAL_MACHINE, R"(SOFTWARE\Valve\Steam)", 0, KEY_QUERY_VALUE | KEY_WOW64_32KEY, &steam) != ERROR_SUCCESS) {
109 return;
110 }
111
112 DWORD steamLocationSize = STEAM_LOCATION_MAX_SIZE;
113 if (RegQueryValueExA(steam, "InstallPath", nullptr, nullptr, reinterpret_cast<LPBYTE>(steamLocationData.get()), &steamLocationSize) != ERROR_SUCCESS) {
114 return;
115 }
116
117 RegCloseKey(steam);
118 steamLocation = steamLocationSize > 0 ? std::string(steamLocationData.get(), steamLocationSize - 1) : "";
119 }
120#else
121 {
122 std::filesystem::path home{std::getenv("HOME")};
123#ifdef __APPLE__
124 steamLocation = home / "Library" / "Application Support" / "Steam";
125#else
126 // Snap install takes priority, the .steam symlink may exist simultaneously with Snap installs
127 steamLocation = home / "snap" / "steam" / "common" / ".steam" / "steam";
128
129 if (!std::filesystem::exists(steamLocation, ec)) {
130 // Use the regular install path
131 steamLocation = home / ".steam" / "steam";
132 }
133#endif
134 }
135
136 if (!std::filesystem::exists(steamLocation, ec)) {
137 std::string location;
138 std::filesystem::path d{"cwd/steamclient64.dll"};
139 for (const auto& entry : std::filesystem::directory_iterator{"/proc/"}) {
140 if (std::filesystem::exists(entry / d, ec)) {
141 ec.clear();
142 const auto s = std::filesystem::read_symlink(entry.path() / "cwd", ec);
143 if (ec) {
144 continue;
145 }
146 location = s.string();
147 break;
148 }
149 }
150 if (location.empty()) {
151 return;
152 }
153 steamLocation = location;
154 }
155#endif
156
157 if (!std::filesystem::exists(steamLocation.string(), ec)) {
158 return;
159 }
160 this->steamInstallDir = steamLocation.string();
161
162 auto libraryFoldersFilePath = steamLocation / "steamapps" / "libraryfolders.vdf";
163 if (!std::filesystem::exists(libraryFoldersFilePath, ec)) {
164 return;
165 }
166
167 KV1 libraryFolders{fs::readFileText(libraryFoldersFilePath.string())};
168
169 const auto& libraryFoldersValue = libraryFolders["libraryfolders"];
170 if (libraryFoldersValue.isInvalid()) {
171 return;
172 }
173
174 for (uint64_t i = 0; i < libraryFoldersValue.getChildCount(); i++) {
175 const auto& folder = libraryFoldersValue[i];
176
177 auto folderName = folder.getKey();
178 if (folderName == "TimeNextStatsReport" || folderName == "ContentStatsID") {
179 continue;
180 }
181
182 const auto& folderPath = folder["path"];
183 if (folderPath.isInvalid()) {
184 continue;
185 }
186
187 std::filesystem::path libraryFolderPath{folderPath.getValue()};
188 {
189 std::string libraryFolderPathProcessedEscapes;
190 bool hitBackslash = false;
191 for (char c : libraryFolderPath.string()) {
192 if (hitBackslash || c != '\\') {
193 libraryFolderPathProcessedEscapes += c;
194 hitBackslash = false;
195 } else {
196 hitBackslash = true;
197 }
198 }
199 libraryFolderPath = libraryFolderPathProcessedEscapes;
200 }
201 libraryFolderPath /= "steamapps";
202
203 this->libraryDirs.push_back(libraryFolderPath.string());
204
205 if (!std::filesystem::exists(libraryFolderPath)) {
206 continue;
207 }
208
209 for (const auto& entry : std::filesystem::directory_iterator{libraryFolderPath, std::filesystem::directory_options::skip_permission_denied}) {
210 auto entryName = entry.path().filename().string();
211 if (!entryName.starts_with("appmanifest_") || !entryName.ends_with(".acf")) {
212 continue;
213 }
214
215 KV1 appManifest(fs::readFileText(entry.path().string()));
216
217 const auto& appState = appManifest["AppState"];
218 if (appState.isInvalid()) {
219 continue;
220 }
221
222 const auto& appName = appState["name"];
223 if (appName.isInvalid()) {
224 continue;
225 }
226 const auto& appInstallDir = appState["installdir"];
227 if (appInstallDir.isInvalid()) {
228 continue;
229 }
230 const auto& appID = appState["appid"];
231 if (appID.isInvalid()) {
232 continue;
233 }
234
235 this->gameDetails[std::stoi(std::string{appID.getValue()})] = GameInfo{
236 .name = std::string{appName.getValue()},
237 .installDir = std::string{appInstallDir.getValue()},
238 .libraryInstallDirsIndex = this->libraryDirs.size() - 1,
239 };
240 }
241 }
