27 [[nodiscard]] std::string_view readFGDString(BufferStreamReadOnly& stream, BufferStream& backing) {
28  parser::text::eatWhitespaceAndSingleLineComments(stream);
29 
30  static constexpr std::string_view END = "\"\n";
31 
32  auto startSpan = backing.tell();
33  while (true) {
34  char c = stream.read<char>();
35  if (c != '\"') {
36  stream.seek(-1, std::ios::cur);
37  auto out = parser::text::readUnquotedStringToBuffer(stream, backing, ":", parser::text::NO_ESCAPE_SEQUENCES);
38  if (stream.seek(-1, std::ios::cur).peek<char>() != ':') {
39  stream.skip();
40  parser::text::eatWhitespaceAndSingleLineComments(stream);
41  }
42  return out;
43  }
44 
45  for (c = stream.read<char>(); END.find(c) == std::string_view::npos; c = stream.read<char>()) {
46  if (c == '\\') {
47  auto n = stream.read<char>();
48  if (n == 'n') {
49  backing << '\n';
50  } else if (END.find(n) != std::string_view::npos) {
51  break;
52  } else {
53  backing << c << n;
54  }
55  } else {
56  backing << c;
57  }
58  }
59 
60  if (!::tryToEatSeparator(stream, '+')) {
61  break;
62  }
63  // We need to make sure the next line is actually a string, because sometimes + will lead to nothing. Lovely
64  parser::text::eatWhitespaceAndSingleLineComments(stream);
65  if (stream.peek<char>() != '\"') {
66  break;
67  }
68  }
69 
70  if (stream.seek(-1, std::ios::cur).peek<char>() != ':') {
71  stream.skip();
72  parser::text::eatWhitespaceAndSingleLineComments(stream);
73  }
74 
75  backing << '\0';
76  return {reinterpret_cast<const char*>(backing.data()) + startSpan, backing.tell() - 1 - startSpan};
77 }

505 void FGD::readEntities(BufferStreamReadOnly& stream, const std::string& path, std::vector<std::string>& seenPaths) {
506  auto& backingString = this->backingData.emplace_back();
507  // Multiply by 2 to ensure buffer will have enough space (very generous)
508  backingString.resize(stream.size() * 2);
509  BufferStream backing{backingString, false};
510 
511  while (true) {
512  parser::text::eatWhitespaceAndSingleLineComments(stream);
513 
514  // All entity definitions start with an '@' followed by the class type
515  if (stream.read<char>() != '@') {
516  throw parser::text::syntax_error{INVALID_SYNTAX_MSG};
517  }
518 
519  auto classType = parser::text::readUnquotedStringToBuffer(stream, backing, "(", parser::text::NO_ESCAPE_SEQUENCES);
520  if (string::iequals(classType, "include")) {
521  parser::text::eatWhitespace(stream);
522  // Assume the include path is relative to the current file being processed
523  auto fgdPath = (std::filesystem::path{path}.parent_path() / parser::text::readStringToBuffer(stream, backing)).string();
524  string::normalizeSlashes(fgdPath);
525  if (std::find(seenPaths.begin(), seenPaths.end(), fgdPath) != seenPaths.end()) {
526  continue;
527  }
528  seenPaths.push_back(fgdPath);
529 
530  auto fgdData = fs::readFileText(fgdPath);
531  if (fgdData.empty()) {
532  continue;
533  }
534 
535  BufferStreamReadOnly newStream{fgdData};
536  try {
537  this->readEntities(newStream, fgdPath, seenPaths);
538  } catch (const std::overflow_error&) {}
539  } else if (string::iequals(classType, "version")) {
540  ::readVersion(stream, backing, this->version);
541  } else if (string::iequals(classType, "mapsize")) {
542  ::readMapSize(stream, backing, this->mapSize);
543  } else if (string::iequals(classType, "MaterialExclusion")) {
544  ::readMaterialExclusionDirs(stream, backing, this->materialExclusionDirs);
545  } else if (string::iequals(classType, "AutoVisGroup")) {
546  ::readAutoVisGroups(stream, backing, this->autoVisGroups);
547  } else if (string::iequals(classType, "BaseClass") ||
548  string::iequals(classType, "PointClass") ||
549  string::iequals(classType, "NPCClass") ||
550  string::iequals(classType, "SolidClass") ||
