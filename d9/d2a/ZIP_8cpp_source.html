93std::optional<std::vector<std::byte>> ZIP::readEntry(const std::string& path_) const {
94 const auto path = this->cleanEntryPath(path_);
95 const auto entry = this->findEntry(path);
96 if (!entry) {
97 return std::nullopt;
98 }
99 if (entry->unbaked) {
100 return readUnbakedEntry(*entry);
101 }
102
103 // It's baked into the file on disk
104 if (!this->streamOpen || !this->zipOpen) {
105 return std::nullopt;
106 }
107 if (mz_zip_locate_entry(this->zipHandle, path.c_str(), !this->isCaseSensitive()) != MZ_OK) {
108 return std::nullopt;
109 }
110 if (mz_zip_entry_read_open(this->zipHandle, 0, nullptr) != MZ_OK) {
111 return std::nullopt;
112 }
113 std::vector<std::byte> out;
114 out.resize(entry->length);
115 mz_zip_entry_read(this->zipHandle, out.data(), static_cast<int>(entry->length));
116 mz_zip_entry_close(this->zipHandle);

249bool ZIP::openZIP(std::string_view path) {
250 this->streamHandle = mz_stream_os_create();
251 if (mz_stream_open(this->streamHandle, path.data(), MZ_OPEN_MODE_READ) != MZ_OK) {
252 mz_stream_delete(&this->streamHandle);
253 return false;
254 }
255 this->streamOpen = true;
256
257 this->zipHandle = mz_zip_create();
258 if (mz_zip_open(this->zipHandle, this->streamHandle, MZ_OPEN_MODE_READ) != MZ_OK) {
259 mz_zip_delete(&this->zipHandle);
260 return false; // No need to delete the stream, it's done when we destruct
261 }
262 this->zipOpen = true;
263
264 return true;
265}
