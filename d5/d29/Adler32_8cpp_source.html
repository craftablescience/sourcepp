43 uint32_t crypto::computeAdler32(const std::byte* buffer, std::size_t len) {
44  uint32_t adler = 0;
45 
46  /* split Adler-32 into component sums */
47  uint32_t sum2 = (adler >> 16) & 0xffff;
48  adler &= 0xffff;
49 
50  /* in case user likes doing a byte at a time, keep it fast */
51  if (len == 1) {
52  adler += static_cast<unsigned char>(buffer[0]);
53  if (adler >= BASE)
54  adler -= BASE;
55  sum2 += adler;
56  if (sum2 >= BASE)
57  sum2 -= BASE;
58  return adler | (sum2 << 16);
59  }
60 
61  /* initial Adler-32 value (deferred check for len == 1 speed) */
62  if (!buffer)
63  return 1L;
64 
65  /* in case short lengths are provided, keep it somewhat fast */
66  if (len < 16) {
67  while (len--) {
68  adler += static_cast<unsigned char>(*buffer++);
69  sum2 += adler;
70  }
71  if (adler >= BASE)
72  adler -= BASE;
73  sum2 %= BASE; /* only added so many BASE's */
74  return adler | (sum2 << 16);
75  }
76 
77  /* do length NMAX blocks -- requires just one modulo operation */
78  while (len >= NMAX) {
79  len -= NMAX;
80  uint32_t n = NMAX / 16; /* NMAX is divisible by 16 */
81  do {
82  DO16(buffer) /* 16 sums unrolled */
83  buffer += 16;
84  } while (--n);
85  adler %= BASE;
86  sum2 %= BASE;
87  }
88 
89  /* do remaining bytes (less than NMAX, still just one modulo) */
90  if (len) { /* avoid modulos if none remaining */
91  while (len >= 16) {
92  len -= 16;
93  DO16(buffer)
94  buffer += 16;
95  }
96  while (len--) {
97  adler += static_cast<unsigned char>(*buffer++);
98  sum2 += adler;
99  }
100  adler %= BASE;
101  sum2 %= BASE;
102  }
103 
104  /* return recombined sums */
105  return adler | (sum2 << 16);
106 }
