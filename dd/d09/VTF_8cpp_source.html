656 void VTF::computeReflectivity() {
657  static constexpr auto getReflectivityForImage = [](VTF& vtf, uint16_t frame, uint8_t face, uint8_t faceCount, uint16_t slice) {
658  static constexpr auto getReflectivityForPixel = [](ImagePixel::RGBA8888* pixel) -> math::Vec3f {
659  // http://markjstock.org/doc/gsd_talk_11_notes.pdf page 11
660  math::Vec3f ref{static_cast<float>(pixel->r), static_cast<float>(pixel->g), static_cast<float>(pixel->b)};
661  ref /= 255.f * 0.9f;
662  ref[0] *= ref[0];
663  ref[1] *= ref[1];
664  ref[2] *= ref[2];
665  return ref;
666  };
667 
668  auto rgba8888Data = vtf.getImageDataAsRGBA8888(0, frame, face, slice);
669  math::Vec3f out{};
670  for (uint64_t i = 0; i < rgba8888Data.size(); i += 4) {
671  out += getReflectivityForPixel(reinterpret_cast<ImagePixel::RGBA8888*>(rgba8888Data.data() + i));
672  }
673  return out / (rgba8888Data.size() / (ImageFormatDetails::bpp(ImageFormat::RGBA8888) / 8));
674  };
675 
676  const auto faceCount = this->getFaceCount();
677 
678 #ifdef SOURCEPP_BUILD_WITH_THREADS
679  if (this->frameCount > 1 || faceCount > 1 || this->sliceCount > 1) {
680  std::vector<std::future<math::Vec3f>> futures;
681  futures.reserve(this->frameCount * faceCount * this->sliceCount);
682 
683  this->reflectivity = {};
684  for (int j = 0; j < this->frameCount; j++) {
685  for (int k = 0; k < faceCount; k++) {
686  for (int l = 0; l < this->sliceCount; l++) {
687  futures.push_back(std::async(std::launch::async, [this, j, k, faceCount, l] {
688  return getReflectivityForImage(*this, j, k, faceCount, l);
689  }));
690  if (std::thread::hardware_concurrency() > 0 && futures.size() >= std::thread::hardware_concurrency() * 2) {
691  for (auto& future : futures) {
692  this->reflectivity += future.get();
693  }
694  futures.clear();
695  }
696  }
697  }
698  }
699 
700  for (auto& future : futures) {
701  this->reflectivity += future.get();
702  }
703  this->reflectivity /= this->frameCount * faceCount * this->sliceCount;
704  } else {
705  this->reflectivity = getReflectivityForImage(*this, 0, 0, 1, 0);
706  }
707 #else
708  this->reflectivity = {};
709  for (int j = 0; j < this->frameCount; j++) {
710  for (int k = 0; k < faceCount; k++) {
711  for (int l = 0; l < this->sliceCount; l++) {
712  this->reflectivity += getReflectivityForImage(*this, j, k, faceCount, l);
713  }
714  }
715  }
716  this->reflectivity /= this->frameCount * faceCount * this->sliceCount;
717 #endif
718 }

763  if (const auto* resource = this->getResource(type); resource && resource->data.size() == data_.size()) {
764  std::memcpy(resource->data.data(), data_.data(), data_.size());
765  return;
766  }
767 
768  // Store resource data
769  std::unordered_map<Resource::Type, std::pair<std::vector<std::byte>, uint64_t>> resourceData;
770  for (const auto& resource : this->resources) {
771  resourceData[resource.type] = {std::vector<std::byte>{resource.data.begin(), resource.data.end()}, 0};
772  }
773 
774  // Set new resource
775  if (data_.empty()) {
776  resourceData.erase(type);
777  } else {
778  resourceData[type] = {{data_.begin(), data_.end()}, 0};
779  }
780 
781  // Save the data
782  this->data.clear();
783  BufferStream writer{this->data};
784 
785  for (auto resourceType : Resource::TYPE_ARRAY_ORDER) {
786  if (!resourceData.contains(resourceType)) {
787  continue;
788  }
789  auto& [specificResourceData, offset] = resourceData[resourceType];
790  if (resourceType == type) {
791  Resource newResource{
792  type,
793  specificResourceData.size() <= sizeof(uint32_t) ? Resource::FLAG_LOCAL_DATA : Resource::FLAG_NONE,
794  {this->data.data() + offset, specificResourceData.size()},
795  };
796  if (auto* resourcePtr = this->getResourceInternal(type)) {
797  *resourcePtr = newResource;
798  } else {
799  this->resources.push_back(newResource);
800  }
801  } else if (!resourceData.contains(resourceType)) {
802  continue;
803  }
804  offset = writer.tell();
805  writer.write(specificResourceData);
806  }
807  this->data.resize(writer.size());
808 
809  for (auto& resource : this->resources) {
810  if (resourceData.contains(resource.type)) {
811  const auto& [specificResourceData, offset] = resourceData[resource.type];
812  resource.data = {this->data.data() + offset, specificResourceData.size()};
813  }
814  }
815 }
