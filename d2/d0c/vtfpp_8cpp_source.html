796 void VTFWriter::computeReflectivity() {
797  this->reflectivity = {};
798  for (int j = 0; j < this->frameCount; j++) {
799  math::Vec3f facesOut{};
800  for (int k = 0; k < this->getFaceCount(); k++) {
801  math::Vec3f slicesOut{};
802  for (int l = 0; l < this->sliceCount; l++) {
803  auto data = this->getImageDataAsRGBA8888(0, j, k, l);
804 
805  math::Vec3f sliceOut{};
806  for (uint64_t i = 0; i < data.size(); i += 4) {
807  // http://markjstock.org/doc/gsd_talk_11_notes.pdf page 11
808  math::Vec3f pixel{static_cast<float>(data[i]), static_cast<float>(data[i+1]), static_cast<float>(data[i+2])};

810  pixel[0] *= pixel[0];
811  pixel[1] *= pixel[1];
812  pixel[2] *= pixel[2];
813  sliceOut += pixel;
814  }
815  sliceOut /= data.size() / 4;
816  slicesOut += sliceOut;
817  }
818  facesOut += slicesOut / this->sliceCount;
819  }
820  this->reflectivity += facesOut / this->getFaceCount();
821  }
822  this->reflectivity /= this->frameCount;

1067  auto headerLengthPos = writer.tell();

1069 
1070  writer
1071  .write(this->width)

1077  .write(this->reflectivity)
1078  .write<uint32_t>(0) // padding
1079  .write(this->bumpMapScale)
1080  .write(this->format)
1081  .write(this->mipCount)
1082  .write(ImageFormat::DXT1)
1083  .write(this->thumbnailWidth)
1084  .write(this->thumbnailHeight);
1085 
1086  if (this->minorVersion >= 2) {
1087  writer << this->sliceCount;
1088  }
1089 
1090  if (this->minorVersion < 3) {
1091  const auto headerAlignment = math::getPaddingForAlignment(16, writer.tell());
1092  for (uint16_t i = 0; i < headerAlignment; i++) {
1093  writer.write<std::byte>({});
1094  }
1095  this->headerSize = writer.tell();
1096  writer.seek_u(headerLengthPos).write<uint32_t>(this->headerSize).seek_u(this->headerSize);
1097 
1098  if (const auto* thumbnailResource = this->getResource(Resource::TYPE_THUMBNAIL_DATA); thumbnailResource && this->hasThumbnailData()) {
1099  writer.write(thumbnailResource->data);
1100  }
1101  if (const auto* imageResource = this->getResource(Resource::TYPE_IMAGE_DATA); imageResource && this->hasImageData()) {
1102  writer.write(imageResource->data);
1103  }
1104  } else {
1105  std::vector<std::byte> auxCompressionResourceData;
1106  std::vector<std::byte> compressedImageResourceData;
1107  bool hasAuxCompression = false;
1108  if (const auto* imageResource = this->getResource(Resource::TYPE_IMAGE_DATA)) {
1109  hasAuxCompression = this->minorVersion >= 6 && this->compressionLevel != 0;
1110  if (hasAuxCompression) {
1111  auxCompressionResourceData.resize((this->mipCount * this->frameCount * this->getFaceCount() + 2) * sizeof(uint32_t));
1112  BufferStream auxWriter{auxCompressionResourceData, false};
1113 
1114  // Format of aux resource is as follows, with each item being a 4 byte integer:
1115  // - Size of resource in bytes, not counting this int
1116  // - Compression level
1117  // - (X times) Size of each mip-face-frame combo
1118  auxWriter
1119  .write<uint32_t>(auxCompressionResourceData.size() - sizeof(uint32_t))
1120  .write<uint32_t>(this->compressionLevel);
1121
