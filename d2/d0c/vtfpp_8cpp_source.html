11 Resource::ConvertedData Resource::convertData() const {
12  switch (this->type) {
13  case TYPE_CRC:
14  case TYPE_EXTENDED_FLAGS:
15  if (this->data.size() != sizeof(uint32_t)) {
16  return {};
17  }
18  return *reinterpret_cast<const uint32_t*>(this->data.data());
19  case TYPE_LOD_CONTROL_INFO:
20  if (this->data.size() != sizeof(uint32_t)) {
21  return {};
22  }
23  return std::make_pair(
24  *(reinterpret_cast<const uint8_t*>(this->data.data()) + 0),
25  *(reinterpret_cast<const uint8_t*>(this->data.data()) + 1));
26  case TYPE_KEYVALUES_DATA:
27  if (this->data.size() <= sizeof(uint32_t)) {
28  return "";
29  }
30  return std::string(reinterpret_cast<const char*>(this->data.data() + 4), *reinterpret_cast<const uint32_t*>(this->data.data()));
31  case TYPE_AUX_COMPRESSION:
32  if (this->data.size() <= sizeof(uint32_t) || this->data.size() % sizeof(uint32_t) != 0) {
33  return {};
34  }
35  return std::span<uint32_t>{reinterpret_cast<uint32_t*>(this->data.data()), this->data.size() / 4};
36  default:
37  break;
38  }
39  return {};
40 }

121  }
122  if (lastResource) {
123  auto offset = *reinterpret_cast<uint32_t*>(lastResource->data.data());
124  auto curPos = stream.tell();
