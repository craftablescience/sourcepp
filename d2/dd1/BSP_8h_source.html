169 [[nodiscard]] auto getLumpData() const {
170 if constexpr (Lump == BSPLump::ENTITIES) {
171 return this->parseEntities();
172 } else if constexpr (Lump == BSPLump::PLANES) {
173 return this->parsePlanes();
174 } else if constexpr (Lump == BSPLump::TEXDATA) {
175 return this->parseTextureData();
176 } else if constexpr (Lump == BSPLump::VERTEXES) {
177 return this->parseVertices();
178 } else if constexpr (Lump == BSPLump::NODES) {
179 return this->parseNodes();
180 } else if constexpr (Lump == BSPLump::TEXINFO) {
181 return this->parseTextureInfo();
182 } else if constexpr (Lump == BSPLump::FACES) {
183 return this->parseFaces();
184 } else if constexpr (Lump == BSPLump::EDGES) {
185 return this->parseEdges();
186 } else if constexpr (Lump == BSPLump::SURFEDGES) {
187 return this->parseSurfEdges();
188 } else if constexpr (Lump == BSPLump::MODELS) {
189 return this->parseBrushModels();
190 } else if constexpr (Lump == BSPLump::ORIGINALFACES) {
191 return this->parseOriginalFaces();
192 } else if constexpr (Lump == BSPLump::GAME_LUMP) {
193 return this->parseGameLumps(true);
194 } else {
195 return this->getLumpData(Lump);
196 }
