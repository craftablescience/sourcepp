777void convertHDRIToCubeMapCPUFallback(std::span<const float> imageDataRGBA32323232F, ImageFormat outputFormat, uint16_t width, uint16_t height, uint16_t resolution, bool bilinear, const std::array<std::array<math::Vec3f, 3>, 6>& startRightUp, std::array<std::vector<std::byte>, 6>& faceData) {
778 for (int i = 0; i < 6; i++) {
779 const math::Vec3f& start = startRightUp[i][0];
780 const math::Vec3f& right = startRightUp[i][1];
781 const math::Vec3f& up = startRightUp[i][2];
782
783 faceData[i].resize(resolution * resolution * sizeof(ImagePixel::RGBA32323232F));
784 std::span<float> face{reinterpret_cast<float*>(faceData[i].data()), reinterpret_cast<float*>(faceData[i].data() + faceData[i].size())};
785
786 for (int row = 0; row < resolution; row++) {
787 for (int col = 0; col < resolution; col++) {
788 math::Vec3f pixelDirection3d{
789 start[0] + ((float) col * 2.f + 0.5f) / (float) resolution * right[0] + ((float) row * 2.f + 0.5f) / (float) resolution * up[0],
790 start[1] + ((float) col * 2.f + 0.5f) / (float) resolution * right[1] + ((float) row * 2.f + 0.5f) / (float) resolution * up[1],
791 start[2] + ((float) col * 2.f + 0.5f) / (float) resolution * right[2] + ((float) row * 2.f + 0.5f) / (float) resolution * up[2],
792 };
793 float azimuth = std::atan2(pixelDirection3d[0], -pixelDirection3d[2]) + math::pi_f32; // add pi to move range to 0-360 deg
794 float elevation = std::atan(pixelDirection3d[1] / std::sqrt(pixelDirection3d[0] * pixelDirection3d[0] + pixelDirection3d[2] * pixelDirection3d[2])) + math::pi_f32 / 2.f;
795 float colHdri = (azimuth / math::pi_f32 / 2.f) * (float) width; // add pi to azimuth to move range to 0-360 deg
796 float rowHdri = (elevation / math::pi_f32) * (float) height;
797 if (!bilinear) {
798 int colNearest = std::clamp((int) colHdri, 0, width - 1);
799 int rowNearest = std::clamp((int) rowHdri, 0, height - 1);
800 face[col * 4 + resolution * row * 4 + 0] = imageDataRGBA32323232F[colNearest * 4 + width * rowNearest * 4 + 0];
801 face[col * 4 + resolution * row * 4 + 1] = imageDataRGBA32323232F[colNearest * 4 + width * rowNearest * 4 + 1];
802 face[col * 4 + resolution * row * 4 + 2] = imageDataRGBA32323232F[colNearest * 4 + width * rowNearest * 4 + 2];
803 face[col * 4 + resolution * row * 4 + 3] = imageDataRGBA32323232F[colNearest * 4 + width * rowNearest * 4 + 3];
804 } else {
805 float intCol, intRow;
806 // factor gives the contribution of the next column, while the contribution of intCol is 1 - factor
807 float factorCol = std::modf(colHdri - 0.5f, &intCol);
808 float factorRow = std::modf(rowHdri - 0.5f, &intRow);
809 int low_idx_row = static_cast<int>(intRow);
810 int low_idx_column = static_cast<int>(intCol);
811 int high_idx_column;
812 if (factorCol < 0.0f) {
813 // modf can only give a negative value if the azimuth falls in the first pixel, left of the
814 // center, so we have to mix with the pixel on the opposite side of the panoramic image
815 high_idx_column = width - 1;
816 } else if (low_idx_column == width - 1) {
817 // if we are in the right-most pixel, and fall right of the center, mix with the left-most pixel
818 high_idx_column = 0;
819 } else {
820 high_idx_column = low_idx_column + 1;
821 }
822 int high_idx_row;
823 if (factorRow < 0.0f) {
824 high_idx_row = height - 1;
825 } else if (low_idx_row == height - 1) {
826 high_idx_row = 0;
827 } else {
828 high_idx_row = low_idx_row + 1;
829 }
830 factorCol = std::abs(factorCol);
831 factorRow = std::abs(factorRow);
832 float f1 = (1 - factorRow) * (1 - factorCol);
833 float f2 = factorRow * (1 - factorCol);
834 float f3 = (1 - factorRow) * factorCol;
835 float f4 = factorRow * factorCol;
836 for (int j = 0; j < 4; j++) {
837 auto interpolatedValue = static_cast<uint8_t>(
838 face[low_idx_column * 4 + width * low_idx_row * 4 + j] * f1 +
839 face[low_idx_column * 4 + width * high_idx_row * 4 + j] * f2 +
840 face[high_idx_column * 4 + width * low_idx_row * 4 + j] * f3 +
841 face[high_idx_column * 4 + width * high_idx_row * 4 + j] * f4
842 );
843 face[col * 4 + resolution * row * 4 + j] = std::clamp<uint8_t>(interpolatedValue, 0, 255);
844 }
845 }
846 }
847 }
848 if (outputFormat != ImageFormat::RGBA32323232F) {
849 faceData[i] = ImageConversion::convertImageDataToFormat(faceData[i], ImageFormat::RGBA32323232F, outputFormat, resolution, resolution);
850 }
851 }

856std::vector<std::byte> ImageConversion::convertImageDataToFormat(std::span<const std::byte> imageData, ImageFormat oldFormat, ImageFormat newFormat, uint16_t width, uint16_t height) {
857 if (imageData.empty() || oldFormat == ImageFormat::EMPTY) {
858 return {};
859 }
860
861 if (oldFormat == newFormat) {
862 return {imageData.begin(), imageData.end()};
863 }
864
865 std::vector<std::byte> newData;
866
867 const ImageFormat intermediaryOldFormat = ImageFormatDetails::containerFormat(oldFormat);
868 if (ImageFormatDetails::compressed(oldFormat)) {
869 newData = ::convertImageDataUsingCompressonator(imageData, oldFormat, intermediaryOldFormat, width, height);
870 } else {
871 switch (intermediaryOldFormat) {
872 case ImageFormat::RGBA8888: newData = ::convertImageDataToRGBA8888(imageData, oldFormat); break;
873 case ImageFormat::RGBA16161616: newData = ::convertImageDataToRGBA16161616(imageData, oldFormat); break;
874 case ImageFormat::RGBA32323232F: newData = ::convertImageDataToRGBA32323232F(imageData, oldFormat); break;
875 default: return {};
876 }
877 }
878
879 if (intermediaryOldFormat == newFormat) {
880 return newData;
881 }
882
883 const ImageFormat intermediaryNewFormat = ImageFormatDetails::containerFormat(newFormat);
884 if (intermediaryOldFormat != intermediaryNewFormat) {
885 if (intermediaryOldFormat == ImageFormat::RGBA8888) {
886 if (intermediaryNewFormat == ImageFormat::RGBA16161616) {
887 newData = ::convertImageDataFromRGBA8888ToRGBA16161616(newData);
888 } else if (intermediaryNewFormat == ImageFormat::RGBA32323232F) {
889 newData = ::convertImageDataFromRGBA8888ToRGBA32323232F(newData);
890 } else {
891 return {};
892 }
893 } else if (intermediaryOldFormat == ImageFormat::RGBA16161616) {
894 if (intermediaryNewFormat == ImageFormat::RGBA8888) {
895 newData = ::convertImageDataFromRGBA16161616ToRGBA8888(newData);
896 } else if (intermediaryNewFormat == ImageFormat::RGBA32323232F) {
897 newData = ::convertImageDataFromRGBA16161616ToRGBA32323232F(newData);
898 } else {
899 return {};
900 }
901 } else if (intermediaryOldFormat == ImageFormat::RGBA32323232F) {
902 if (intermediaryNewFormat == ImageFormat::RGBA8888) {
903 newData = ::convertImageDataFromRGBA32323232FToRGBA8888(newData);
904 } else if (intermediaryNewFormat == ImageFormat::RGBA16161616) {
905 newData = ::convertImageDataFromRGBA32323232FToRGBA16161616(newData);
906 } else {
907 return {};
908 }
909 } else {
910 return {};
911 }
912 }
913
914 if (intermediaryNewFormat == newFormat) {
915 return newData;
916 }
917
918 if (ImageFormatDetails::compressed(newFormat)) {
919 newData = ::convertImageDataUsingCompressonator(newData, intermediaryNewFormat, newFormat, width, height);
920 } else {
921 switch (intermediaryNewFormat) {
922 case ImageFormat::RGBA8888: newData = ::convertImageDataFromRGBA8888(newData, newFormat); break;
923 case ImageFormat::RGBA16161616: newData = ::convertImageDataFromRGBA16161616(newData, newFormat); break;
924 case ImageFormat::RGBA32323232F: newData = ::convertImageDataFromRGBA32323232F(newData, newFormat); break;
925 default: return {};
926 }
927 }
928
929 return newData;

1299
1300 // EXR
1301 if (EXRVersion version; ParseEXRVersionFromMemory(&version, reinterpret_cast<const unsigned char*>(fileData.data()), fileData.size()) == TINYEXR_SUCCESS) {
1302 if (version.multipart || version.non_image) {
1303 return {};
1304 }
1305
1306 EXRHeader header;
1307 InitEXRHeader(&header);
1308 const char* err = nullptr;
1309 if (ParseEXRHeaderFromMemory(&header, &version, reinterpret_cast<const unsigned char*>(fileData.data()), fileData.size(), &err) != TINYEXR_SUCCESS) {
1310 FreeEXRErrorMessage(err);
1311 return {};
1312 }
1313
1314 // Sanity check
1315 if (header.num_channels < 1) {
1316 FreeEXRHeader(&header);
1317 return {};
1318 }
1319
1320 // Define the channel names we support (RGBA, greyscale)
1321 std::unordered_map<std::string_view, int> channelIndices{{"R", -1}, {"G", -1}, {"B", -1}, {"A", -1}, {"Y", -1}};
1322
1323 // Get channel type (EXR supports different types per channel, we do not)
1324 // Rather than bailing we ask EXR to convert the lowest precision data
1325 auto channelType = header.pixel_types[0];
1326 for (int i = 1; i < header.num_channels; i++) {
1327 // UINT -> HALF -> FLOAT
1328 if (header.pixel_types[i] > channelType && channelIndices.contains(header.channels[i].name)) {
1329 channelType = header.pixel_types[i];
1330 }
1331 }

1333 if (channelType == TINYEXR_PIXELTYPE_UINT) {
1334 channelType = TINYEXR_PIXELTYPE_HALF;
1335 }

1338 for (int i = 0; i < header.num_channels; i++) {
1339 if (channelIndices.contains(header.channels[i].name)) {
1340 channelIndices[header.channels[i].name] = i;
1341 }

1344 if (channelIndices["A"] >= 0) {

1346 } else {
1347 if (channelType == TINYEXR_PIXELTYPE_HALF) {
1348 // VTF has no RGB161616F
1349 channelType = TINYEXR_PIXELTYPE_FLOAT;
1350 }
1351 format = ImageFormat::RGB323232F;
1352 }
1353 channelIndices["R"] = channelIndices["Y"];
1354 channelIndices["G"] = channelIndices["Y"];
1355 channelIndices["B"] = channelIndices["Y"];
1356 } else if (channelIndices["A"] >= 0) {
1357 format = channelType == TINYEXR_PIXELTYPE_HALF ? ImageFormat::RGBA16161616F : ImageFormat::RGBA32323232F;
1358 } else if (channelIndices["B"] >= 0) {
1359 if (channelType == TINYEXR_PIXELTYPE_HALF) {
1360 // VTF has no RGB161616F
1361 channelType = TINYEXR_PIXELTYPE_FLOAT;
1362 }
1363 format = ImageFormat::RGB323232F;
1364 } else if (channelIndices["G"] >= 0) {
1365 format = channelType == TINYEXR_PIXELTYPE_HALF ? ImageFormat::RG1616F : ImageFormat::RG3232F;
1366 } else if (channelIndices["R"] >= 0) {
1367 format = channelType == TINYEXR_PIXELTYPE_HALF ? ImageFormat::R16F : ImageFormat::R32F;
1368 } else {
1369 FreeEXRHeader(&header);
1370 return {};
1371 }
1372
1373 // Now that channelType has stopped changing, we can set it properly
1374 for (int i = 0; i < header.num_channels; i++) {
1375 if (header.pixel_types[i] != channelType && channelIndices.contains(header.channels[i].name)) {
1376 header.requested_pixel_types[i] = channelType;
1377 }
1378 }
1379
1380 EXRImage image;
1381 InitEXRImage(&image);
1382 if (LoadEXRImageFromMemory(&image, &header, reinterpret_cast<const unsigned char*>(fileData.data()), fileData.size(), &err) != TINYEXR_SUCCESS) {
1383 FreeEXRErrorMessage(err);
1384 FreeEXRHeader(&header);
1385 return {};
1386 }
1387
1388 width = image.width;
1389 height = image.height;
