393  if (outputDir.empty() || !predicate) {
394  return false;
395  }
396 
397  // Get list of paths
398  std::vector<Entry> saveEntries;
399  for (const auto& [dir, bakedEntries_] : this->getBakedEntries()) {
400  for (const auto& entry : bakedEntries_) {
401  if (predicate(entry)) {
402  saveEntries.push_back(entry);
403  }
404  }
405  }
406  for (const auto& [dir, unbakedEntries_] : this->getUnbakedEntries()) {
407  for (const auto& entry : unbakedEntries_) {
408  if (predicate(entry)) {
409  saveEntries.push_back(entry);
410  }
411  }
412  }
413  if (saveEntries.empty()) {
414  return false;
415  }
416 
417  // Strip shared directories until we have a root folder
418  std::vector<std::string> rootDirList;
419  {
420  std::vector<std::vector<std::string>> pathSplits;
421  for (const auto& entry : saveEntries) {
422  pathSplits.push_back(::splitPath(entry.path));
423  }
424  while (true) {
425  bool allTheSame = true;
426  const std::string& first = pathSplits[0][0];
427  for (const auto& path : pathSplits) {
428  if (path.size() == 1) {
429  allTheSame = false;
430  break;

432  if (path[0] != first) {
433  allTheSame = false;
434  break;
435  }
436  }
437  if (!allTheSame) {
438  break;
439  }
440  rootDirList.push_back(first);

444  }
445  }

447

450  bool noneFailed = true;
451  for (const auto& entry : saveEntries) {
452  std::string entryPath = entry.path.substr(rootDirLen);
