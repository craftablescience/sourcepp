433  std::vector<std::string> saveEntryPaths;
434  this->runForAllEntries([&predicate, &saveEntryPaths](const std::string& path, const Entry& entry) {
435  if (predicate(path, entry)) {
436  saveEntryPaths.push_back(path);
437  }
438  });
439  if (saveEntryPaths.empty()) {
440  return false;
441  }
442

444  if (stripSharedDirs) {
445  // Strip shared directories until we have a root folder
446  std::vector<std::string> rootDirList;

449  pathSplits.reserve(saveEntryPaths.size());

452  }

468  }

472  }

475  }
