55void replace(std::string& line, const std::string& oldString, const std::string& newString) {
56 const auto oldSize = oldString.length();
57 if (oldSize > line.length()) {
58 return;
59 }
60
61 const auto newSize = newString.length();
62 std::size_t pos = 0;
63 while (true) {
64 pos = line.find(oldString, pos);
65 if (pos == std::string::npos) {
66 break;
67 }
68 if (oldSize == newSize) {
69 line.replace(pos, oldSize, newString);
70 } else {
71 line.erase(pos, oldSize);
72 line.insert(pos, newString);
73 }
74 pos += newSize;
75 }
76}

78void fixFilePathForWindows(std::string& path) {
79 // Remove invalid characters
80 ::replace(path, "<", "_");
81 ::replace(path, "<", "_");
82 ::replace(path, ">", "_");
83 ::replace(path, ":", "_");
84 ::replace(path, "\"", "_");
85 ::replace(path, "|", "_");
86 ::replace(path, "?", "_");
87 ::replace(path, "*", "_");
88
89 std::filesystem::path filePath{path};
90 auto filename = filePath.filename().string();
91 auto extension = filePath.extension().string();
92 auto stem = filePath.stem().string();
93 string::toUpper(stem);
94
95 // Replace bad filenames
96 if (stem == "CON" || stem == "PRN" || stem == "AUX" || stem == "NUL") {
97 filename = "___" + extension;
98 } else if (stem.length() == 4 && stem[3] != '0' && ((stem.starts_with("COM") || stem.starts_with("LPT")))) {
99 filename = "___";
100 filename += stem[3];
101 filename += extension;
102 }
103
104 // Files cannot end with a period - weird
105 if (extension == ".") {
106 filename.pop_back();
107 filename += '_';
108 }
109
110 path = (filePath.parent_path() / filename).string();
111}

427bool PackFile::extractAll(const std::string& outputDir, const EntryPredicate& predicate, bool stripSharedDirs) const {
428 if (outputDir.empty() || !predicate) {
429 return false;
430 }
431
432 // Get list of paths
433 std::vector<std::string> saveEntryPaths;
434 this->runForAllEntries([&predicate, &saveEntryPaths](const std::string& path, const Entry& entry) {
435 if (predicate(path, entry)) {
436 saveEntryPaths.push_back(path);
437 }
438 });
439 if (saveEntryPaths.empty()) {
440 return false;
441 }
442
443 std::size_t rootDirLen = 0;
444 if (stripSharedDirs) {
445 // Strip shared directories until we have a root folder
446 std::vector<std::string> rootDirList;
447
448 std::vector<std::vector<std::string>> pathSplits;
449 pathSplits.reserve(saveEntryPaths.size());
450 for (const auto& path : saveEntryPaths) {
451 pathSplits.push_back(::splitPath(path));
452 }
453 while (true) {
454 bool allTheSame = true;
455 const std::string& first = pathSplits[0][0];
456 for (const auto& path : pathSplits) {
457 if (path.size() == 1) {
458 allTheSame = false;
459 break;
460 }
461 if (path[0] != first) {
462 allTheSame = false;
463 break;
464 }
465 }
466 if (!allTheSame) {
467 break;
468 }
469 rootDirList.push_back(first);
470 for (auto& path : pathSplits) {
471 path.erase(path.begin());
472 }
473 }
474 rootDirLen = ::joinPath(rootDirList).length() + 1;
475 }
476
477 // Extract
478 std::filesystem::path outputDirPath{outputDir};
479 bool noneFailed = true;
480 for (const auto& path : saveEntryPaths) {
481 auto savePath = path;
482#ifdef _WIN32
483 ::fixFilePathForWindows(savePath);
484#endif
485 if (!this->extractEntry(path, (outputDirPath / savePath.substr(rootDirLen)).string())) {
486 noneFailed = false;
487 }
488 }
489 return noneFailed;
490}
