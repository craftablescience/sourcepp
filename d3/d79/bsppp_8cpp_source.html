42BSP::BSP(std::string path_)
43 : path(std::move(path_)) {
44 FileStream reader{this->path};
45 if (!reader) {
46 this->path.clear();
47 return;
48 }
49
50 if (auto signature = reader.seek_in(0).read<uint32_t>(); signature != BSP_SIGNATURE) {
51 // File is not a BSP
52 this->path.clear();
53 return;
54 }
55 this->header.version = reader.read<int32_t>();
56
57 // Contagion funny
58 if (this->header.version == 27) {
59 reader.skip_in<int32_t>();
60 }
61
62 reader >> this->header.lumps;
63
64 // If no offsets are larger than 1024 (less than the size of the BSP header, but greater than any lump version),
65 // it's probably a L4D2 BSP and those are lump versions!
66 if (this->header.version == 21) {
67 int i;
68 for (i = 0; i < BSP_LUMP_COUNT; i++) {
69 if (this->header.lumps[i].offset > 1024) {
70 break;
71 }
72 }
73 this->isL4D2 = i == BSP_LUMP_COUNT;
74 if (this->isL4D2) {
75 // Swap fields around
76 for (i = 0; i < BSP_LUMP_COUNT; i++) {
77 std::swap(this->header.lumps[i].offset, this->header.lumps[i].version);
78 std::swap(this->header.lumps[i].offset, this->header.lumps[i].length);
79 }
80 }
81 }
82
83 reader >> this->header.mapRevision;
84}

150void BSP::writeLump(BSPLump lumpIndex, std::span<const std::byte> data, bool condenseFile) {
151 if (this->path.empty() || lumpIndex == BSPLump::UNKNOWN) {
152 return;
153 }
154
155 auto lumpToMove = static_cast<std::underlying_type_t<BSPLump>>(lumpIndex);
156
157 if (!this->hasLump(lumpIndex) || !condenseFile) {
158 // Put the lump at the end of the file
159 int32_t lastLumpOffset = 0, lastLumpLength = 0;
160 for (const auto& [lumpOffset, lumpLength, lumpVersion, lumpFourCC] : this->header.lumps) {
161 if (lastLumpOffset < this->header.lumps[lumpToMove].offset) {
162 lastLumpOffset = lumpOffset;
163 lastLumpLength = lumpLength;
164 }
165 }
166 if (lastLumpOffset == 0) {
167 // Whole file is full of empty lumps
168 this->header.lumps[lumpToMove].offset = sizeof(Header);
169 } else {
170 this->header.lumps[lumpToMove].offset = lastLumpOffset + lastLumpLength;
171 }
172 this->header.lumps[lumpToMove].length = static_cast<int32_t>(data.size());
173 } else {
174 // Sort lumps by file position
175 std::array<int, BSP_LUMP_COUNT> lumpIDs{};
176 for (int i = 0; i < lumpIDs.size(); i++) {
177 lumpIDs[i] = i;
178 }
179 std::sort(lumpIDs.begin(), lumpIDs.end(), [this](int lhs, int rhs) {
180 return this->header.lumps[lhs].offset < this->header.lumps[rhs].offset;
181 });
182
183 // Condense the lumps in the order they appear in the file, and move the new lump to the end
184 FileStream bsp{this->path, FileStream::OPT_READ | FileStream::OPT_WRITE};
185 int32_t currentOffset = 0;
186 for (int i = 0; i < lumpIDs.size(); i++) {
187 auto lumpID = lumpIDs[i];
188
189 if (lumpID == lumpToMove) {
190 continue;
191 }
192 if (!currentOffset) {
193 currentOffset = this->header.lumps[lumpID].offset + this->header.lumps[lumpID].length;
194 continue;
195 }
196
197 auto lumpsData = bsp.seek_in(this->header.lumps[lumpID].offset).read_bytes(this->header.lumps[lumpID].length);
198 bsp.seek_out(currentOffset).write(lumpsData);
199
200 this->header.lumps[lumpID].offset = currentOffset;
201 currentOffset += this->header.lumps[lumpID].length;
202
203 // If we have the space to add padding (we should), then do so
204 // This should never fail for well-constructed BSP files
205 auto padding = math::paddingForAlignment(4, currentOffset);
206 if (padding && i < lumpIDs.size() - 1 && currentOffset + padding <= this->header.lumps[lumpIDs[i + 1]].offset) {
207 currentOffset += padding;
208 }
209 }
210
211 this->header.lumps[lumpToMove].offset = currentOffset;
212 this->header.lumps[lumpToMove].length = static_cast<int32_t>(data.size());
213 }
214
215 // Write modified header and lump
216 this->writeHeader();
217 {
218 FileStream writer{this->path, FileStream::OPT_READ | FileStream::OPT_WRITE};
219 writer.seek_out(this->header.lumps[lumpToMove].offset).write(data);
220 }
221
222 // Resize file if it shrank
223 int32_t lastLumpOffset = 0, lastLumpLength = 0;
224 for (const auto& [lumpOffset, lumpLength, lumpVersion, lumpFourCC] : this->header.lumps) {
225 if (lastLumpOffset < this->header.lumps[lumpToMove].offset) {
226 lastLumpOffset = lumpOffset;
227 lastLumpLength = lumpLength;
228 }
229 }
230 if (std::filesystem::file_size(this->path) > lastLumpOffset + lastLumpLength) {
231 std::filesystem::resize_file(this->path, lastLumpOffset + lastLumpLength);
232 }
233}

235bool BSP::applyLumpPatchFile(const std::string& lumpFilePath) {
236 if (this->path.empty()) {
237 return false;
238 }
239
240 FileStream reader{lumpFilePath};
241 if (!reader) {
242 return false;
243 }
244
245 const auto offset = reader.read<int32_t>();
246 const auto index = reader.read<int32_t>();
247 const auto version = reader.read<int32_t>();
248 const auto length = reader.read<int32_t>();
249 if (index < 0 || index > BSP_LUMP_COUNT || offset <= 0 || length <= 0) {
250 return false;
251 }
252
253 this->header.lumps[index].version = version;
254 this->writeLump(static_cast<BSPLump>(index), reader.seek_in(offset).read_bytes(length));
255 return true;

258void BSP::createLumpPatchFile(BSPLump lumpIndex) const {
259 auto lumpData = this->readLump(lumpIndex);
260 if (!lumpData) {
261 return;
262 }
263
264 const auto& [lumpOffset, lumpLength, lumpVersion, lumpFourCC] = this->header.lumps.at(static_cast<std::underlying_type_t<BSPLump>>(lumpIndex));
265
266 const auto fsPath = std::filesystem::path{this->path};
267 const auto fsStem = (fsPath.parent_path() / fsPath.stem()).string() + "_l_";
268 int nonexistentNumber;
269 for (nonexistentNumber = 0; true; nonexistentNumber++) {
270 if (!std::filesystem::exists(fsStem + std::to_string(nonexistentNumber) + ".lmp")) {
271 break;
272 }
273 }
274
275 FileStream writer{fsStem + std::to_string(nonexistentNumber) + ".lmp", FileStream::OPT_TRUNCATE | FileStream::OPT_CREATE_IF_NONEXISTENT};
276 writer
277 .seek_out(0)
278 .write<int32_t>(sizeof(int32_t) * 5)
279 .write(lumpIndex)
280 .write(lumpVersion)
281 .write(lumpLength)
282 .write(this->header.mapRevision)
283 .write(*lumpData);

286void BSP::writeHeader() const {
287 FileStream writer{this->path, FileStream::OPT_READ | FileStream::OPT_WRITE};
288 writer.seek_out(0) << BSP_SIGNATURE << this->header.version;
289
290 // Contagion funny
291 if (this->header.version == 27) {
292 writer.write<int32_t>(0);
293 }
294
295 if (!this->isL4D2) {
296 writer << this->header.lumps;
297 } else {
298 for (int i = 0; i < static_cast<std::underlying_type_t<BSPLump>>(BSPLump::COUNT); i++) {
299 writer << this->header.lumps[i].version << this->header.lumps[i].offset << this->header.lumps[i].length << this->header.lumps[i].fourCC;
300 }
301 }
302
303 writer << this->header.mapRevision;
304}
