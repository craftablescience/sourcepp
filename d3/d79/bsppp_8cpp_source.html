201 void BSP::moveLumpToWritableSpace(BSPLump lumpIndex, int32_t newSize) {
202  auto lumpToMove = static_cast<uint32_t>(lumpIndex);
203  this->header.lumps[lumpToMove].length = newSize;
204 
205  // If the lump doesn't exist, put it at the end of the file
206  if (!this->hasLump(lumpIndex)) {
207  int lastLumpOffset = 0, lastLumpLength = 0;
208  for (const Lump& lump : this->header.lumps) {
209  if (lastLumpOffset < this->header.lumps[lumpToMove].offset) {
210  lastLumpOffset = lump.offset;
211  lastLumpLength = lump.length;
212  }
213  }
214  if (lastLumpOffset == 0) {
215  // Whole file is full of empty lumps
216  this->header.lumps[lumpToMove].offset = sizeof(Header);
217  } else {
218  this->header.lumps[lumpToMove].offset = lastLumpOffset + lastLumpLength;
219  }
220  this->header.lumps[lumpToMove].length = newSize;
221 
222  this->writeHeader();
223  return;
224  }
225 
226  // If the moving lump is at the end of the file we just overwrite it, otherwise we have to shift some lumps over
227  std::vector<int> lumpsAfterMovingLumpIndices;
228  for (int i = 0; i < BSP_LUMP_COUNT; i++) {
229  if (this->header.lumps[i].offset > this->header.lumps[lumpToMove].offset) {
230  lumpsAfterMovingLumpIndices.push_back(i);
231  }
232  }
233  if (lumpsAfterMovingLumpIndices.empty()) {
234  return;
235  }
236 
237  // Get the exact area to move
238  int moveOffsetStart = std::numeric_limits<int>::max(), moveOffsetEnd = 0;
239  for (int lumpsAfterMovingLumpIndex : lumpsAfterMovingLumpIndices) {
240  if (this->header.lumps[lumpsAfterMovingLumpIndex].offset < moveOffsetStart) {
241  moveOffsetStart = this->header.lumps[lumpsAfterMovingLumpIndex].offset;
242  }
243  if (auto offsetAndLength = this->header.lumps[lumpsAfterMovingLumpIndex].offset + this->header.lumps[lumpsAfterMovingLumpIndex].length; offsetAndLength > moveOffsetEnd) {
244  moveOffsetEnd = offsetAndLength;
245  }
246  }
247 
248  // Get where to move it
249  int lastLumpBeforeMovingLumpOffset = 0, lastLumpBeforeMovingLumpLength = 0;
250  for (const Lump& lump : this->header.lumps) {
251  if (lump.offset < this->header.lumps[lumpToMove].offset && lump.offset > lastLumpBeforeMovingLumpOffset) {
252  lastLumpBeforeMovingLumpOffset = lump.offset;
253  lastLumpBeforeMovingLumpLength = lump.length;
254  }
255  }
256 
257  // Move all the lumps after paklump back
258  {
259  FileStream bsp{this->path, FileStream::OPT_READ | FileStream::OPT_WRITE};
260  auto lumpsData = bsp.seek_in(moveOffsetStart).read_bytes(moveOffsetEnd - moveOffsetStart);
261  bsp.seek_out(lastLumpBeforeMovingLumpOffset + lastLumpBeforeMovingLumpLength).write(lumpsData);
262 
263  // Fix the offsets
264  for (int lumpsAfterMovingLumpIndex : lumpsAfterMovingLumpIndices) {
265  this->header.lumps[lumpsAfterMovingLumpIndex].offset -= newSize;
266  }
267  this->header.lumps[lumpToMove].offset = lastLumpBeforeMovingLumpOffset + lastLumpBeforeMovingLumpLength + static_cast<int>(lumpsData.size());
268  }
269 
270  // Write modified header
271  this->writeHeader();
272 }
