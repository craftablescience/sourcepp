149 void BSP::writeLump(BSPLump lumpIndex, std::span<const std::byte> data, bool condenseFile) {
150  if (this->path.empty() || lumpIndex == BSPLump::UNKNOWN) {
151  return;
152  }
153 
154  auto lumpToMove = static_cast<std::underlying_type_t<BSPLump>>(lumpIndex);
155 
156  if (!this->hasLump(lumpIndex) || !condenseFile) {
157  // Put the lump at the end of the file
158  int32_t lastLumpOffset = 0, lastLumpLength = 0;
159  for (const Lump& lump : this->header.lumps) {
160  if (lastLumpOffset < this->header.lumps[lumpToMove].offset) {
161  lastLumpOffset = lump.offset;
162  lastLumpLength = lump.length;
163  }
164  }
165  if (lastLumpOffset == 0) {
166  // Whole file is full of empty lumps
167  this->header.lumps[lumpToMove].offset = sizeof(Header);
168  } else {
169  this->header.lumps[lumpToMove].offset = lastLumpOffset + lastLumpLength;
170  }
171  this->header.lumps[lumpToMove].length = static_cast<int32_t>(data.size());
172  } else {
173  // Sort lumps by file position
174  std::array<int, BSP_LUMP_COUNT> lumpIDs{};
175  for (int i = 0; i < lumpIDs.size(); i++) {
176  lumpIDs[i] = i;
177  }
178  std::sort(lumpIDs.begin(), lumpIDs.end(), [this](int lhs, int rhs) {
179  return this->header.lumps[lhs].offset < this->header.lumps[rhs].offset;
180  });
181 
182  // Condense the lumps in the order they appear in the file, and move the new lump to the end
183  FileStream bsp{this->path, FileStream::OPT_READ | FileStream::OPT_WRITE};
184  int32_t currentOffset = 0;
185  for (int i = 0; i < lumpIDs.size(); i++) {
186  auto lumpID = lumpIDs[i];
187 
188  if (lumpID == lumpToMove) {
189  continue;
190  }
191  if (!currentOffset) {
192  currentOffset = this->header.lumps[lumpID].offset + this->header.lumps[lumpID].length;
193  continue;
194  }
195 
196  auto lumpsData = bsp.seek_in(this->header.lumps[lumpID].offset).read_bytes(this->header.lumps[lumpID].length);
197  bsp.seek_out(currentOffset).write(lumpsData);
198 
199  this->header.lumps[lumpID].offset = currentOffset;
200  currentOffset += this->header.lumps[lumpID].length;
201 
202  // If we have the space to add padding (we should), then do so
203  // This should never fail for well-constructed BSP files
204  auto padding = math::getPaddingForAlignment(4, currentOffset);
205  if (padding && i < lumpIDs.size() - 1 && currentOffset + padding <= this->header.lumps[lumpIDs[i + 1]].offset) {
206  currentOffset += padding;
207  }
208  }
209 
210  this->header.lumps[lumpToMove].offset = currentOffset;
211  this->header.lumps[lumpToMove].length = static_cast<int32_t>(data.size());
212  }
213 
214  // Write modified header and lump
215  this->writeHeader();
216  {
217  FileStream writer{this->path, FileStream::OPT_READ | FileStream::OPT_WRITE};
218  writer.seek_out(this->header.lumps[lumpToMove].offset).write(data);
219  }
220 
221  // Resize file if it shrank
222  int32_t lastLumpOffset = 0, lastLumpLength = 0;
223  for (const Lump& lump : this->header.lumps) {
224  if (lastLumpOffset < this->header.lumps[lumpToMove].offset) {
225  lastLumpOffset = lump.offset;
226  lastLumpLength = lump.length;
227  }
228  }
229  if (std::filesystem::file_size(this->path) > lastLumpOffset + lastLumpLength) {
230  std::filesystem::resize_file(this->path, lastLumpOffset + lastLumpLength);
231  }
