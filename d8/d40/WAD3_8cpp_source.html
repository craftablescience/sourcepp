66std::unique_ptr<PackFile> WAD3::open(const std::string& path, const EntryCallback& callback) {
67 if (!std::filesystem::exists(path)) {
68 // File does not exist
69 return nullptr;
70 }
71
72 auto* wad = new WAD3{path};
73 std::unique_ptr<PackFile> packFile{wad};
74
75 FileStream reader{wad->fullFilePath};
76 reader.seek_in(0);
77
78 // Verify the identity
79 if (auto signature = reader.read<uint32_t>(); signature != WAD3_SIGNATURE) {
80 return nullptr;
81 }
82
83 // Treating counts as unsigned to simplify some logic... No reason to really have a negative count here anyways
84 auto lumpCount = reader.read<uint32_t>();
85 auto lumpOffset = reader.read<uint32_t>();
86
87 // Read in all lump entries
88 reader.seek_in(lumpOffset);
89 for (uint32_t i = 0; i < lumpCount; i++) {
90 // Read all entry data
91 auto offset = reader.read<int32_t>();
92 auto size = reader.read<int32_t>();
93 reader.skip_in<int32_t>(); // size_uncompressed
94 auto type = reader.read<int8_t>();
95 reader.skip_in<int8_t>(); // compression
96 reader.skip_in<int16_t>(); // padding
97 auto name = reader.read_string(WAD3_FILENAME_MAX_SIZE);
98
99 // We'll append the type onto the name as an extension
100 if(type >= WFT_FIRST && type < WFT_COUNT) {
101 name += ".";
102 name += k_FileTypeNames[type - WFT_FIRST];
103 }
104
105 // Create the entry
106 Entry entry = createNewEntry();
107 entry.offset = offset;
108 entry.length = size;
109 entry.unbaked = false;
110
111 auto entryPath = wad->cleanEntryPath(name);
112 wad->entries.emplace(entryPath, entry);
113
114 if (callback) {
115 callback(entryPath, entry);
116 }
117 }
118
119 return packFile;
120}

123std::optional<std::vector<std::byte>> WAD3::readEntry(const std::string& path_) const {
124 auto path = this->cleanEntryPath(path_);
125 auto entry = this->findEntry(path);
126 if (!entry) {
127 return std::nullopt;
128 }
129
130 // We have it in memory, no need to read from disk
131 if (entry->unbaked) {
132 return readUnbakedEntry(*entry);
133 }
134
135 // It's baked into the file on disk. Open it, read it, send it back
136 FileStream stream{this->fullFilePath};
137 if (!stream) {
138 return std::nullopt;
139 }
140 stream.seek_in_u(entry->offset);
141 return stream.read_bytes(entry->length);
142}

151bool WAD3::bake(const std::string& outputDir_, BakeOptions options, const EntryCallback& callback) {
152 // Get the proper file output folder
153 std::string outputDir = this->getBakeOutputDir(outputDir_);
154 std::string outputPath = outputDir + '/' + this->getFilename();
155
156 // File output stream
157 FileStream stream{outputPath, FileStream::OPT_TRUNCATE | FileStream::OPT_CREATE_IF_NONEXISTENT};
158 stream.seek_out(0);
159
160 // File data to be appended at the end
161 std::vector<std::byte> fileData;
162
163 // We'll quickly tally up and preallocate our vector so we don't need to do extra allocations at run time
164 uint32_t totalDataLength = 0;
165 this->runForAllEntries([&totalDataLength](const std::string&, const Entry& entry) {
166 totalDataLength += entry.length;
167 });
168 fileData.reserve(totalDataLength);
169
170 // Counter to keep track of where we're writing data to, starts after the header
171 uint32_t currentOffset = sizeof(uint32_t) * 3;
172
173 // Write out the header
174 stream.write(WAD3_SIGNATURE);
175 stream.write<uint32_t>(this->getEntryCount());
176 stream.write<uint32_t>(currentOffset);
177
178 // Push forward to end of lump info array
179 currentOffset += this->getEntryCount() * 32;
180
181 this->runForAllEntriesInternal([this, &callback, &stream, &fileData, currentOffset](std::string path, Entry& entry) {
182 // Append the lump data to the end of the data vector
183 if (auto binData = this->readEntry(path)) {
184 entry.offset = currentOffset + fileData.size();
185 fileData.insert(fileData.end(), binData->begin(), binData->end());
186 }
187 else {
188 entry.offset = 0;
189 entry.length = 0;
190 }
191
192 // Convert the extension back into the type
193 int type = 0;
194 const std::size_t pos = path.find_last_of('.');
195 if (pos > 0) {
196 const std::string_view ext = path.c_str() + pos + 1;
197 for (int i = WFT_FIRST; i < WFT_COUNT; i++) {
198 if (string::iequals(ext, k_FileTypeNames[i - WFT_FIRST])) {
199 type = i;
200 break;
201 }
202 }
203 // Chop off the extension
204 path = path.substr(0, pos);
205 }
206
207 // Write the lump header
208 stream.write<uint32_t>(entry.offset);
209 stream.write<uint32_t>(0);
210 stream.write<uint32_t>(entry.length);
211 stream.write<uint8_t>(type); // type
212 stream.write<uint8_t>(0); // compression
213 stream.write<uint16_t>(0); // padding
214 stream.write(path, false, WAD3_FILENAME_MAX_SIZE);
215
216 if (callback) {
217 callback(path, entry);
218 }
219 });
220
221 // File data
222 stream.write(fileData);
223
224 // Clean up
225 this->mergeUnbakedEntries();
226 PackFile::setFullFilePath(outputDir);
227 return true;
228}
