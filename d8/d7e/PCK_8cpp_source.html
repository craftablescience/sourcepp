48std::unique_ptr<PackFile> PCK::open(const std::string& path, const EntryCallback& callback) {
49 if (!std::filesystem::exists(path)) {
50 // File does not exist
51 return nullptr;
52 }
53
54 auto* pck = new PCK{path};
55 auto packFile = std::unique_ptr<PackFile>(pck);
56
57 FileStream reader{pck->fullFilePath};
58 reader.seek_in(0);
59
60 if (auto signature = reader.read<uint32_t>(); signature != PCK_SIGNATURE) {
61 // PCK might be embedded
62 reader.seek_in(sizeof(uint32_t), std::ios::end);
63 if (auto endSignature = reader.read<uint32_t>(); endSignature != PCK_SIGNATURE) {
64 return nullptr;
65 }
66
67 reader.seek_in(-static_cast<int64_t>(sizeof(uint32_t) + sizeof(uint64_t)), std::ios::cur);
68 auto distanceIntoFile = reader.read<uint64_t>();
69
70 reader.seek_in(-static_cast<int64_t>(distanceIntoFile + sizeof(uint64_t)), std::ios::cur);
71 if (auto startSignature = reader.read<uint32_t>(); startSignature != PCK_SIGNATURE) {
72 return nullptr;
73 }
74
75 pck->startOffset = reader.tell_in() - sizeof(uint32_t);
76 }
77
78 reader.read(pck->header.packVersion);
79 reader.read(pck->header.godotVersionMajor);
80 reader.read(pck->header.godotVersionMinor);
81 reader.read(pck->header.godotVersionPatch);
82
83 pck->header.flags = FLAG_NONE;
84 std::size_t extraEntryContentsOffset = 0;
85 if (pck->header.packVersion > 1) {
86 pck->header.flags = reader.read<FlagsV2>();
87 extraEntryContentsOffset = reader.read<uint64_t>();
88 }
89
90 if (pck->header.flags & FLAG_ENCRYPTED) {
91 // File directory is encrypted
92 return nullptr;
93 }
94 if (pck->header.flags & FLAG_RELATIVE_FILE_DATA) {
95 extraEntryContentsOffset += pck->startOffset;
96 pck->header.flags = static_cast<FlagsV2>(pck->header.flags & ~FLAG_RELATIVE_FILE_DATA);
97 }
98
99 // Reserved
100 reader.skip_in<int32_t>(16);
101
102 // Directory
103 auto fileCount = reader.read<uint32_t>();
104 for (uint32_t i = 0; i < fileCount; i++) {
105 Entry entry = createNewEntry();
106
107 auto entryPath = pck->cleanEntryPath(reader.read_string(reader.read<uint32_t>()));
108 if (entryPath.starts_with(PCK_PATH_PREFIX)) {
109 entryPath = entryPath.substr(PCK_PATH_PREFIX.length());
110 }
111
112 entry.offset = reader.read<uint64_t>() + extraEntryContentsOffset;
113 entry.length = reader.read<uint64_t>();
114 entry.extraData = reader.read_bytes(16);
115
116 if (pck->header.packVersion > 1) {
117 entry.flags = reader.read<uint32_t>();
118 }
119
120 pck->entries.emplace(entryPath, entry);
121
122 if (callback) {
123 callback(entryPath, entry);
124 }
125 }
126
127 // File data
128 pck->dataOffset = reader.tell_in();
129
130 return packFile;
131}

133std::optional<std::vector<std::byte>> PCK::readEntry(const std::string& path_) const {
134 auto path = this->cleanEntryPath(path_);

150 if (!stream) {
151 return std::nullopt;

167bool PCK::bake(const std::string& outputDir_, BakeOptions options, const EntryCallback& callback) {
168 // Get the proper file output folder
169 std::string outputDir = this->getBakeOutputDir(outputDir_);
170 std::string outputPath = outputDir + '/' + this->getFilename();
171
172 // Reconstruct data for ease of access
173 std::vector<std::pair<std::string, Entry*>> entriesToBake;
174 this->runForAllEntriesInternal([&entriesToBake](const std::string& path, Entry& entry) {
175 entriesToBake.emplace_back(path, &entry);
176 });
177
178 // Read data before overwriting, we don't know if we're writing to ourself
179 std::vector<std::byte> fileData;
180 for (auto& [path, entry] : entriesToBake) {
181 if (auto binData = this->readEntry(path)) {
182 entry->offset = fileData.size();
183
184 fileData.insert(fileData.end(), binData->begin(), binData->end());
185 const auto padding = math::paddingForAlignment(PCK_FILE_DATA_PADDING, static_cast<int>(entry->length));
186 for (int i = 0; i < padding; i++) {
187 fileData.push_back(static_cast<std::byte>(0));
188 }
189 } else {
190 entry->offset = 0;
191 entry->length = 0;
192 }
193 }
194
195 // If this is an embedded pck, read the executable data first
196 std::vector<std::byte> exeData;
197 if (this->startOffset > 0) {
198 FileStream stream{this->fullFilePath};
199 if (!stream) {
200 return false;
201 }
202 stream.seek_in(0);
203 exeData = stream.read_bytes(this->startOffset);
204 }
205
206 // Write data
207 {
208 FileStream stream{outputPath, FileStream::OPT_TRUNCATE | FileStream::OPT_CREATE_IF_NONEXISTENT};
209 stream.seek_out(0);
210
211 if (!exeData.empty()) {
212 stream.write(exeData);
213 }
214
215 // Signature
216 stream.write(PCK_SIGNATURE);
217
218 // Header
219 stream.write(this->header.packVersion);
220 stream.write(this->header.godotVersionMajor);
221 stream.write(this->header.godotVersionMinor);
222 stream.write(this->header.godotVersionPatch);
223
224 if (this->header.packVersion > 1) {
225 stream.write(this->header.flags);
226 stream.write<uint64_t>(0);
227 }
228
229 // Reserved
230 stream.write(std::array<int32_t, 16>{});
231
232 // Directory start
233 stream.write(static_cast<uint32_t>(entriesToBake.size()));
234
235 // Dry-run to get the length of the directory section
236 this->dataOffset = stream.tell_out();
237 for (const auto& path : std::views::keys(entriesToBake)) {
238 const auto entryPath = std::string{PCK_PATH_PREFIX} + path;
239 const auto padding = math::paddingForAlignment(PCK_DIRECTORY_STRING_PADDING, static_cast<int>(entryPath.length()));
240 this->dataOffset +=
241 sizeof(uint32_t) + // Path length
242 entryPath.length() + padding + // Path
243 (sizeof(std::size_t) * 2) + // Offset, Length
244 (sizeof(std::byte) * 16); // MD5
245
246 if (this->header.packVersion > 1) {
247 this->dataOffset += sizeof(uint32_t); // Flags
248 }
249 }
250
251 // Directory
252 for (const auto& [path, entry] : entriesToBake) {
253 const auto entryPath = std::string{PCK_PATH_PREFIX} + path;
254 const auto padding = math::paddingForAlignment(PCK_DIRECTORY_STRING_PADDING, static_cast<int>(entryPath.length()));
255 stream.write(static_cast<uint32_t>(entryPath.length() + padding));
256 stream.write(entryPath, false, entryPath.length() + padding);
257
258 entry->offset += this->dataOffset;
259 stream.write(entry->offset);
260 stream.write(entry->length);
261 stream.write(entry->extraData);
262
263 if (this->header.packVersion > 1) {
264 stream.write(entry->flags);
265 }
266
267 if (callback) {
268 callback(path, *entry);
269 }
270 }
271
272 // File data
273 stream.write(fileData);
274
275 // Write offset to start
276 if (this->startOffset > 0) {
277 stream.write(stream.tell_out() - startOffset);
278 stream.write(PCK_SIGNATURE);
279 }
280 }
281
282 // Clean up
283 this->mergeUnbakedEntries();
284 PackFile::setFullFilePath(outputDir);
285 return true;
286}
