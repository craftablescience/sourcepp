120bool GMA::bake(const std::string& outputDir_, BakeOptions options, const EntryCallback& callback) {
121 // Get the proper file output folder
122 std::string outputDir = this->getBakeOutputDir(outputDir_);
123 std::string outputPath = outputDir + '/' + this->getFilename();
124
125 // Reconstruct data for ease of access
126 std::vector<std::pair<std::string, Entry*>> entriesToBake;
127 this->runForAllEntriesInternal([&entriesToBake](const std::string& path, Entry& entry) {
128 entriesToBake.emplace_back(path, &entry);
129 });
130
131 // Read data before overwriting, we don't know if we're writing to ourself
132 std::vector<std::byte> fileData;
133 for (auto& [path, entry] : entriesToBake) {
134 if (auto binData = this->readEntry(path)) {
135 fileData.insert(fileData.end(), binData->begin(), binData->end());
136 } else {
137 entry->length = 0;
138 }
139 }
140
141 {
142 FileStream stream{outputPath, FileStream::OPT_TRUNCATE | FileStream::OPT_CREATE_IF_NONEXISTENT};
143 stream.seek_out(0);
144
145 // Header
146 stream.write(this->header.signature);
147 stream.write(this->header.version);
148 stream.write(this->header.steamID);
149 stream.write(this->header.timestamp);
150 stream.write(this->header.requiredContent);
151 stream.write(this->header.addonName);
152 stream.write(this->header.addonDescription);
153 stream.write(this->header.addonAuthor);
154 stream.write(this->header.addonVersion);
155
156 // File tree
157 for (uint32_t i = 1; i <= entriesToBake.size(); i++) {
158 stream.write(i);
159 const auto& [path, entry] = entriesToBake[i - 1];
160 stream.write(path);
161 stream.write(entry->length);
162 stream.write<uint32_t>(options.gma_writeCRCs ? entry->crc32 : 0);
163
164 if (callback) {
165 callback(path, *entry);
166 }
167 }
168 stream.write<uint32_t>(0);
169
170 // Fix offsets
171 std::size_t offset = stream.tell_out();
172 for (auto& [path, entry] : entriesToBake) {
173 entry->offset = offset;
174 offset += entry->length;
175 }
176
177 // File data
178 stream.write(fileData);
179 }
180
181 // CRC of everything that's been written
182 uint32_t crc = 0;
183 if (options.gma_writeCRCs) {
184 auto fileSize = std::filesystem::file_size(outputPath);
185 FileStream stream{outputPath};
186 stream.seek_in(0);
187 crc = crypto::computeCRC32(stream.read_bytes(fileSize));
188 }
189 {
190 FileStream stream{outputPath, FileStream::OPT_APPEND};
191 stream.write(crc);
192 }
193
194 // Clean up
195 this->mergeUnbakedEntries();
196 PackFile::setFullFilePath(outputDir);
197 return true;
198}
