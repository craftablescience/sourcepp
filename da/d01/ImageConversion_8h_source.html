397template<ImagePixel::PixelType P>
398[[nodiscard]] std::vector<std::byte> extractChannelFromImageData(std::span<const std::byte> imageData, auto P::*channel) {
399 using C = sourcepp::member_type_t<decltype(channel)>;
400 if (imageData.empty() || imageData.size() % sizeof(P) != 0) {
401 return {};
402 }
403
404 std::span pixels{reinterpret_cast<const P*>(imageData.data()), imageData.size() / sizeof(P)};
405
406 std::vector<std::byte> out(imageData.size() / sizeof(P) * sizeof(C));
407 BufferStream stream{out, false};
408 for (const auto& pixel : pixels) {
409 stream << pixel.*channel;
410 }
411 return out;
412}
413
418template<ImagePixel::PixelType P>
419bool applyChannelToImageData(std::span<std::byte> imageData, std::span<const std::byte> channelData, auto P::*channel) {
420 using C = sourcepp::member_type_t<decltype(channel)>;
421 if (imageData.empty() || imageData.size() % sizeof(P) != 0 || channelData.empty() || channelData.size() % sizeof(C) != 0 || imageData.size() / sizeof(P) != channelData.size() / sizeof(C)) {
422 return false;
423 }
424
425 std::span pixels{reinterpret_cast<P*>(imageData.data()), imageData.size() / sizeof(P)};
426 std::span values{reinterpret_cast<C*>(channelData.data()), channelData.size() / sizeof(C)};
427
428 for (int i = 0; i < pixels.size(); i++) {
429 pixels[i].*channel = values[i];
430 }
431 return true;
432}
