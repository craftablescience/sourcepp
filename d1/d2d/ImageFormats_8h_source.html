657[[nodiscard]] constexpr uint8_t getRecommendedMipCountForDims(ImageFormat format, uint16_t width, uint16_t height) {
658 // Note that compressed formats need 4x4 pixel blocks
659 if (sourcepp::math::isPowerOf2(width) && sourcepp::math::isPowerOf2(height)) {
660 const auto log2 = std::bit_width(width > height ? height : width);
661 if (!ImageFormatDetails::compressed(format)) {
662 return log2;
663 }
664 // Eliminate 2x2, 1x1
665 return log2 - 2 > 1 ? log2 - 2 : 1;
666 }
667
668 uint8_t maxMipCount = 1;
669 if (ImageFormatDetails::compressed(format)) {
670 while (width > 0 && height > 0 && width % 4 == 0 && height % 4 == 0) {
671 width /= 2;
672 height /= 2;
673 maxMipCount++;
674 }
675 } else {
676 while (width > 0 && height > 0 && width % 2 == 0 && height % 2 == 0) {
677 width /= 2;
678 height /= 2;
679 maxMipCount++;
680 }
681 }
682 return maxMipCount;
683}

685[[nodiscard]] constexpr uint8_t getActualMipCountForDimsOnConsole(uint16_t width, uint16_t height) {
686 if (width == 0 || height == 0) {
687 return 0;
688 }
689 uint8_t numMipLevels = 1;
690 while (true) {
691 if (width == 1 && height == 1) {
692 break;
693 }
694 width >>= 1;
695 if (width < 1) {
696 width = 1;
697 }
698 height >>= 1;
699 if (height < 1) {
700 height = 1;
701 }
702 numMipLevels += 1;
703 }
704 return numMipLevels;
705}

711[[nodiscard]] constexpr uint32_t getDataLength(ImageFormat format, uint16_t width, uint16_t height, uint16_t sliceCount = 1) {
712 switch(format) {
713 using enum ImageFormat;
714 case DXT3:
715 case DXT5:
716 case ATI2N:
717 case BC7:
718 case BC6H:
719 case ATI1N:
720 case DXT1:
721 case DXT1_ONE_BIT_ALPHA:
722 return ((((width + 3) / 4) < 1) ? 1 : ((width + 3) / 4)) * ((((height + 3) / 4) < 1) ? 1 : ((height + 3) / 4)) * sliceCount * bpp(format) * 2;
723 default:
724 break;
725 }
726 return width * height * sliceCount * (bpp(format) / 8);
727}

729[[nodiscard]] constexpr uint32_t getDataLength(ImageFormat format, uint8_t mipCount, uint16_t frameCount, uint8_t faceCount, uint16_t width, uint16_t height, uint16_t sliceCount = 1) {
730 uint32_t length = 0;
731 for (int mip = mipCount - 1; mip >= 0; mip--) {
732 length += ImageFormatDetails::getDataLength(format, ImageDimensions::getMipDim(mip, width), ImageDimensions::getMipDim(mip, height), sliceCount) * frameCount * faceCount;
733 }
734 return length;

737[[nodiscard]] constexpr bool getDataPosition(uint32_t& offset, uint32_t& length, ImageFormat format, uint8_t mip, uint8_t mipCount, uint16_t frame, uint16_t frameCount, uint8_t face, uint8_t faceCount, uint16_t width, uint16_t height, uint16_t slice = 0, uint16_t sliceCount = 1) {
738 offset = 0;
739 length = 0;
740 for (int i = mipCount - 1; i >= 0; i--) {
741 for (int j = 0; j < frameCount; j++) {
742 for (int k = 0; k < faceCount; k++) {
743 for (int l = 0; l < sliceCount; l++) {
744 const auto imageSize = ImageFormatDetails::getDataLength(format, ImageDimensions::getMipDim(i, width), ImageDimensions::getMipDim(i, height));
745 if (i == mip && j == frame && k == face && l == slice) {
746 length = imageSize;
747 return true;
748 } else {
749 offset += imageSize;
750 }
751 }
752 }
753 }
754 }
755 return false;
