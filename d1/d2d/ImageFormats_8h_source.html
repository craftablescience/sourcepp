538[[nodiscard]] constexpr uint8_t getRecommendedMipCountForDims(ImageFormat format, uint16_t width, uint16_t height) {
539 // Note that compressed formats need 4x4 pixel blocks
540 if (sourcepp::math::isPowerOf2(width) && sourcepp::math::isPowerOf2(height)) {
541 const auto log2 = std::bit_width(width > height ? height : width);
542 if (!ImageFormatDetails::compressed(format)) {
543 return log2;
544 }
545 // Eliminate 2x2, 1x1
546 return log2 - 2 > 1 ? log2 - 2 : 1;
547 }
548
549 uint8_t maxMipCount = 1;
550 if (ImageFormatDetails::compressed(format)) {
551 while (width > 0 && height > 0 && width % 4 == 0 && height % 4 == 0) {
552 width /= 2;
553 height /= 2;
554 maxMipCount++;
555 }
556 } else {
557 while (width > 0 && height > 0 && width % 2 == 0 && height % 2 == 0) {
558 width /= 2;
559 height /= 2;
560 maxMipCount++;
561 }
562 }
563 return maxMipCount;
564}

570[[nodiscard]] constexpr uint32_t getDataLength(ImageFormat format, uint16_t width, uint16_t height, uint16_t sliceCount = 1) {
571 switch(format) {
572 using enum ImageFormat;
573 case DXT3:
574 case DXT5:
575 case ATI2N:
576 case BC7:
577 case BC6H:
578 case ATI1N:
579 case DXT1:
580 case DXT1_ONE_BIT_ALPHA:
581 return ((((width + 3) / 4) < 1) ? 1 : ((width + 3) / 4)) * ((((height + 3) / 4) < 1) ? 1 : ((height + 3) / 4)) * sliceCount * bpp(format) * 2;
582 default:
583 break;
584 }
585 return width * height * sliceCount * (bpp(format) / 8);
586}

588[[nodiscard]] constexpr uint32_t getDataLength(ImageFormat format, uint8_t mipCount, uint16_t frameCount, uint8_t faceCount, uint16_t width, uint16_t height, uint16_t sliceCount = 1) {
589 uint32_t length = 0;
590 for (int mip = mipCount - 1; mip >= 0; mip--) {
591 length += ImageFormatDetails::getDataLength(format, ImageDimensions::getMipDim(mip, width), ImageDimensions::getMipDim(mip, height), sliceCount) * frameCount * faceCount;
592 }

596[[nodiscard]] constexpr bool getDataPosition(uint32_t& offset, uint32_t& length, ImageFormat format, uint8_t mip, uint8_t mipCount, uint16_t frame, uint16_t frameCount, uint8_t face, uint8_t faceCount, uint16_t width, uint16_t height, uint16_t slice = 0, uint16_t sliceCount = 1) {
597 offset = 0;
598 length = 0;
599 for (int i = mipCount - 1; i >= 0; i--) {
600 for (int j = 0; j < frameCount; j++) {
601 for (int k = 0; k < faceCount; k++) {
602 for (int l = 0; l < sliceCount; l++) {
603 const auto imageSize = ImageFormatDetails::getDataLength(format, ImageDimensions::getMipDim(i, width), ImageDimensions::getMipDim(i, height));
604 if (i == mip && j == frame && k == face && l == slice) {
605 length = imageSize;
606 return true;
607 } else {
608 offset += imageSize;
609 }
610 }
611 }
612 }
613 }
614 return false;
