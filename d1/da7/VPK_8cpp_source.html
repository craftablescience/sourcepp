95  if (!std::filesystem::exists(path)) {
96  // File does not exist
97  return nullptr;
98  }
99

101  auto packFile = std::unique_ptr<PackFile>(vpk);
102 
103  FileStream reader{vpk->fullFilePath};
104  reader.seek_in(0);
105  reader.read(vpk->header1);
106  if (vpk->header1.signature != VPK_SIGNATURE) {
107  // File is not a VPK
108  return nullptr;
109  }
110  if (vpk->header1.version == 2) {
111  reader.read(vpk->header2);
112  } else if (vpk->header1.version != 1) {
113  // Apex Legends, Titanfall, etc. are not supported
114  return nullptr;
115  }
116 
117  // Extensions
118  while (true) {
119  std::string extension;
120  reader.read(extension);
121  if (extension.empty())
122  break;
123 
124  // Directories
125  while (true) {
126  std::string directory;
127  reader.read(directory);
128  if (directory.empty())
129  break;
130 
131  std::string fullDir;
132  if (directory == " ") {
133  fullDir = "";
134  } else {
135  fullDir = directory;
136  }
137 
138  // Files
139  while (true) {
140  std::string entryName;
141  reader.read(entryName);
142  if (entryName.empty())
143  break;
144

146 
147  std::string entryPath;
148  if (extension == " ") {
149  entryPath = fullDir.empty() ? "" : fullDir + '/';
150  entryPath += entryName;
151  } else {

153  entryPath += entryName + '.';
154  entryPath += extension;
155  }

159  auto preloadedDataSize = reader.read<uint16_t>();
160  entry.archiveIndex = reader.read<uint16_t>();
161  entry.offset = reader.read<uint32_t>();
162  entry.length = reader.read<uint32_t>();
163 
164  if (reader.read<uint16_t>() != VPK_ENTRY_TERM) {
165  // Invalid terminator!
166  return nullptr;
167  }
168 
169  if (preloadedDataSize > 0) {
170  entry.extraData = reader.read_bytes(preloadedDataSize);
171  entry.length += preloadedDataSize;
172  }
173 
174  if (entry.archiveIndex != VPK_DIR_INDEX && entry.archiveIndex > vpk->numArchives) {
175  vpk->numArchives = entry.archiveIndex;
176  }
177 
178  vpk->entries.emplace(entryPath, entry);
179 
180  if (callback) {
181  callback(entryPath, entry);
182  }
183  }
184  }
185  }
186 
187  // If there are no archives, -1 will be incremented to 0
188  vpk->numArchives++;
189 
190  // Read VPK2-specific data
191  if (vpk->header1.version != 2)
192  return packFile;
193 
194  // Skip over file data, if any
195  reader.seek_in(vpk->header2.fileDataSectionSize, std::ios::cur);
196 
197  if (vpk->header2.archiveMD5SectionSize % sizeof(MD5Entry) != 0)
198  return nullptr;
199 
200  vpk->md5Entries.clear();

202  for (unsigned int i = 0; i < entryNum; i++)
203  vpk->md5Entries.push_back(reader.read<MD5Entry>());
204 
205  if (vpk->header2.otherMD5SectionSize != 48)
206  // This should always be 48
207  return packFile;
208 
209  vpk->footer2.treeChecksum = reader.read_bytes<16>();
210  vpk->footer2.md5EntriesChecksum = reader.read_bytes<16>();
211  vpk->footer2.wholeFileChecksum = reader.read_bytes<16>();
212 
213  if (!vpk->header2.signatureSectionSize) {
214  return packFile;
215  }
216 
217  auto publicKeySize = reader.read<int32_t>();
218  if (vpk->header2.signatureSectionSize == 20 && publicKeySize == VPK_SIGNATURE) {
219  // CS2 beta VPK, ignore it
220  return packFile;
221  }
222 
223  vpk->footer2.publicKey = reader.read_bytes(publicKeySize);
224  vpk->footer2.signature = reader.read_bytes(reader.read<int32_t>());

335 void VPK::addEntryInternal(Entry& entry, const std::string& path, std::vector<std::byte>& buffer, EntryOptions options) {
336  entry.crc32 = crypto::computeCRC32(buffer);
337  entry.length = buffer.size();
338 
339  // Offset will be reset when it's baked, assuming we're not replacing an existing chunk (when flags = 1)
340  entry.flags = 0;
341  entry.offset = 0;
342  entry.archiveIndex = options.vpk_saveToDirectory ? VPK_DIR_INDEX : this->numArchives;
343  if (!options.vpk_saveToDirectory && !this->freedChunks.empty()) {
344  int64_t bestChunkIndex = -1;
345  std::size_t currentChunkGap = SIZE_MAX;
346  for (int64_t i = 0; i < this->freedChunks.size(); i++) {
347  if (
348  (bestChunkIndex < 0 && this->freedChunks[i].length >= entry.length) ||
349  (bestChunkIndex >= 0 && this->freedChunks[i].length >= entry.length && (this->freedChunks[i].length - entry.length) < currentChunkGap)
350  ) {
351  bestChunkIndex = i;
352  currentChunkGap = this->freedChunks[i].length - entry.length;
353  }
354  }
355  if (bestChunkIndex >= 0) {
356  entry.flags |= VPK_FLAG_REUSING_CHUNK;
357  entry.offset = this->freedChunks[bestChunkIndex].offset;
358  entry.archiveIndex = this->freedChunks[bestChunkIndex].archiveIndex;
359  this->freedChunks.erase(this->freedChunks.begin() + bestChunkIndex);
360  if (currentChunkGap < SIZE_MAX && currentChunkGap > 0) {
361  // Add the remaining free space as a free chunk
362  this->freedChunks.push_back({entry.offset + entry.length, currentChunkGap, entry.archiveIndex});
363  }
364  }
365  }
366 
367  if (options.vpk_preloadBytes > 0) {
368  auto clampedPreloadBytes = std::clamp(options.vpk_preloadBytes, 0u, buffer.size() > VPK_MAX_PRELOAD_BYTES ? VPK_MAX_PRELOAD_BYTES : static_cast<uint32_t>(buffer.size()));
369  entry.extraData.resize(clampedPreloadBytes);
370  std::memcpy(entry.extraData.data(), buffer.data(), clampedPreloadBytes);
371  buffer.erase(buffer.begin(), buffer.begin() + clampedPreloadBytes);
372  }
373 
374  // Now that archive index is calculated for this entry, check if it needs to be incremented
375  if (!options.vpk_saveToDirectory && !(entry.flags & VPK_FLAG_REUSING_CHUNK)) {
376  entry.offset = this->currentlyFilledChunkSize;
377  this->currentlyFilledChunkSize += static_cast<int>(buffer.size());
378  if (this->currentlyFilledChunkSize > this->chunkSize) {
379  this->currentlyFilledChunkSize = 0;
380  this->numArchives++;
381  }
382  }

599  }
600  {
601  wholeFileChecksumMD5.Update(reinterpret_cast<const CryptoPP::byte*>(this->md5Entries.data()), this->md5Entries.size() * sizeof(MD5Entry));
602  CryptoPP::Weak::MD5 md5EntriesChecksumMD5;
603  md5EntriesChecksumMD5.Update(reinterpret_cast<const CryptoPP::byte*>(this->md5Entries.data()), this->md5Entries.size() * sizeof(MD5Entry));
604  md5EntriesChecksumMD5.Final(reinterpret_cast<CryptoPP::byte*>(this->footer2.md5EntriesChecksum.data()));
605  }
606  wholeFileChecksumMD5.Update(reinterpret_cast<const CryptoPP::byte*>(this->footer2.treeChecksum.data()), this->footer2.treeChecksum.size());
607  wholeFileChecksumMD5.Update(reinterpret_cast<const CryptoPP::byte*>(this->footer2.md5EntriesChecksum.data()), this->footer2.md5EntriesChecksum.size());
608  wholeFileChecksumMD5.Final(reinterpret_cast<CryptoPP::byte*>(this->footer2.wholeFileChecksum.data()));
609 
610  // We can't recalculate the signature without the private key
611  this->footer2.publicKey.clear();
612  this->footer2.signature.clear();
613  }
614 
615  // Write new headers
616  outDir.seek_out(0);
617  outDir.write(this->header1);
618 
619  // v2 adds the MD5 hashes and file signature
620  if (this->header1.version != 2) {
621  PackFile::setFullFilePath(outputDir);
622  return true;
623  }
624 
625  outDir.write(this->header2);
626 
627  // Add MD5 hashes
628  outDir.seek_out_u(sizeof(Header1) + sizeof(Header2) + this->header1.treeSize + dirVPKEntryData.size());
629  outDir.write(this->md5Entries);
630  outDir.write(this->footer2.treeChecksum);
631  outDir.write(this->footer2.md5EntriesChecksum);
632  outDir.write(this->footer2.wholeFileChecksum);
633 
634  // The signature section is not present
635  PackFile::setFullFilePath(outputDir);
636  return true;
637 }

710 bool VPK::sign(const std::vector<std::byte>& privateKey, const std::vector<std::byte>& publicKey) {
711  if (this->header1.version != 2) {
712  return false;
713  }
714 
715  this->header2.signatureSectionSize = this->footer2.publicKey.size() + this->footer2.signature.size() + sizeof(uint32_t) * 2;
716  {
717  FileStream stream{this->getFilepath().data(), FileStream::OPT_READ | FileStream::OPT_WRITE};
718  stream.seek_out(sizeof(Header1));
719  stream.write(this->header2);
720  }
721 
722  auto dirFileBuffer = fs::readFileBuffer(this->getFilepath().data());
723  if (dirFileBuffer.size() <= this->header2.signatureSectionSize) {
724  return false;
725  }
726  for (int i = 0; i < this->header2.signatureSectionSize; i++) {
727  dirFileBuffer.pop_back();
728  }
729  this->footer2.publicKey = publicKey;
730  this->footer2.signature = crypto::signDataWithSHA256PrivateKey(dirFileBuffer, privateKey);
731 
732  {
733  FileStream stream{this->getFilepath().data(), FileStream::OPT_READ | FileStream::OPT_WRITE};
734  stream.seek_out(this->getHeaderLength() + this->header1.treeSize + this->header2.fileDataSectionSize + this->header2.archiveMD5SectionSize + this->header2.otherMD5SectionSize);
735  stream.write(static_cast<uint32_t>(this->footer2.publicKey.size()));
736  stream.write(this->footer2.publicKey);
737  stream.write(static_cast<uint32_t>(this->footer2.signature.size()));
738  stream.write(this->footer2.signature);
739  }
740  return true;
741 }
